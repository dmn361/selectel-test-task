## Задание 4

```Dockerfile
FROM ubuntu:latest
# нельзя использовать тег latest из-за возможных конфликтов во время
# обновления образа в регистри, нужно задать конкретную версию.
# для оптимизации можно использовать сразу nginx:alpine,
# тогда образ будет из двух слоёв (FROM и COPY), без установки пакетов
# но если нам принципиально использовать ubuntu, то ниже ошибки по докерфайлу с ubuntu
MAINTAINER MyCompany
# устаревшая инструкция, можно использовать label
COPY . /var/www/html
# можно перенести ниже, добавить пользователя после установки пакетов 
# и использовать с флагом --chown=user:user для безопасности 
# (не запускать контейнер с root-пользователем по умолчанию),
# написать под это дело Dockerignore для того, чтобы не тянуть лишние файлы
RUN apt-get update -y
RUN apt-get install -y nginx
# можно было выполнить одним слоем и добавить очистку кеша

CMD ["nginx", "-g", "daemon off;"]
```
